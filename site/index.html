Intent DSL Example

AIM: weather#intent@2.2

INTENT WeatherLookup {
  SUMMARY: "Real-time weather lookup with error handling."

  REQUIREMENTS {
    - "User enters a city name to view temperature."
    - "UI displays error if city is not found."
  }

  CONTRACT FetchWeather {
    INPUT { city: string required }
    ENSURES {
      - "CALLS External.WeatherAPI"
      - "RETURNS WeatherSnapshot"
    }
  }
}
